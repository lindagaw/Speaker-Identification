1function vad_flag = M2FED_vadG729(speech, VAD_cst_param)
  2% VADG729 Implement the Voice Activity Detection Algorithm.
  3% Note that although G.729 VAD operates on pre-processed speech data, this
  4% function is a standalone version, i.e. the pre-processing (highpass
  5% filtering and linear predictive analysis) is also included.
  6%
  7% This function is in support of the 'G.729 Voice Activity Detection'
  8% example and may change in a future release
  9
 10% Copyright 2015 The MathWorks, Inc.
 11
 12%% Algorithm Components Initialization
 13persistent HPF AC LEV1 LEV2 LPC2LSF ZCD VAD_var_param
 14
 15if isempty(HPF)
 16    % Create a IIR digital filter used for pre-processing
 17    HPF = dsp.BiquadFilter('SOSMatrix', VAD_cst_param.HPF_sos);                    
 18                    
 19    % Create an autocorrelator and set its properties to compute the lags 
 20    % in the range [0:NP].
 21    AC = dsp.Autocorrelator('MaximumLagSource', 'Property', ...
 22                            'MaximumLag', VAD_cst_param.NP);
 23                        
 24    % Create a Levinson solver which compute the reflection coefficients 
 25    % from auto-correlation function using the Levinson-Durbin recursion. 
 26    % The first object is configured to output polynomial coefficients and 
 27    % the second object is configured to output reflection coefficients.
 28    LEV1 = dsp.LevinsonSolver('AOutputPort', true, ...
 29                             'KOutputPort', false);
 30
 31    LEV2 = dsp.LevinsonSolver('AOutputPort', false, ...
 32                              'KOutputPort', true);
 33                         
 34    % Create a converter from linear prediction coefficients (LPC) to line 
 35    % spectral frequencies (LSF)
 36    LPC2LSF = dsp.LPCToLSF;    
 37    
 38    % Create a zero crossing detector
 39    ZCD = dsp.ZeroCrossingDetector;
 40    
 41    % initialize variable parameters
 42    VAD_var_param.window_buffer = single(zeros(VAD_cst_param.L_WINDOW - VAD_cst_param.L_FRAME, 1));
 43    VAD_var_param.frm_count = single(0);
 44    VAD_var_param.MeanLSF = single(zeros(VAD_cst_param.M, 1));
 45    VAD_var_param.MeanSE = single(0);
 46    VAD_var_param.MeanSLE = single(0);
 47    VAD_var_param.MeanE = single(0);
 48    VAD_var_param.MeanSZC = single(0);
 49    VAD_var_param.count_sil = single(0);
 50    VAD_var_param.count_update = single(0);
 51    VAD_var_param.count_ext = single(0);
 52    VAD_var_param.less_count = single(0);
 53    VAD_var_param.flag = single(1);
 54    VAD_var_param.prev_markers = single([1, 1]);
 55    VAD_var_param.prev_energy = single(0);
 56    VAD_var_param.Prev_Min = single(Inf);
 57    VAD_var_param.Next_Min = single(Inf);
 58    VAD_var_param.Min_buffer = single(Inf * ones(1, VAD_cst_param.No));    
 59end
 60
 61%% Constants Initialization
 62NOISE = single(0);
 63VOICE = single(1);
 64v_flag = single(0);%#ok
 65VAD_var_param.frm_count = single(VAD_var_param.frm_count + 1);
 66
 67%% Pre-processing
 68% Filter the speech frame: this high-pass filter serves as a precaution 
 69% against undesired low-frequency components.
 70speech_hp = step(HPF, 32768*speech);
 71
 72% Store filtered data to the pre-processed speech buffer
 73speech_buf = [VAD_var_param.window_buffer; speech_hp];
 74
 75% LPC analysis
 76
 77% Windowing of signal
 78speech_win = VAD_cst_param.hamwindow .* speech_buf;
 79
 80% Autocorrelation
 81r = step(AC, speech_win) .* VAD_cst_param.lagwindow;
 82
 83% LSF
 84a = step(LEV1, r(1:VAD_cst_param.M+1));
 85LSF = step(LPC2LSF, a) / (2 * pi);
 86
 87% Reflection coefficients
 88rc = step(LEV2, r(1:3));
 89
 90%% VAD starts here
 91
 92%% Parameters extraction
 93
 94% Full-band energy
 95Ef = 10 * log10(r(1) / VAD_cst_param.L_WINDOW);
 96
 97% Low-band energy
 98El = r(1) * VAD_cst_param.lbf_corr(1) + 2 * sum(r(2:end) .* VAD_cst_param.lbf_corr(2:end));
 99El = 10 * log10(El / VAD_cst_param.L_WINDOW);
100
101% Spectral Distorsion
102SD = sum((LSF-VAD_var_param.MeanLSF).^2);
103
104% Zero-crossing rate
105idx = VAD_cst_param.L_WINDOW - VAD_cst_param.L_NEXT - VAD_cst_param.L_FRAME + 1;
106ZC = double(step(ZCD, speech_buf(idx:idx+VAD_cst_param.L_FRAME)))/VAD_cst_param.L_FRAME;
107
108% Long-term minimum energy
109VAD_var_param.Next_Min = min(Ef, VAD_var_param.Next_Min);
110Min = min(VAD_var_param.Prev_Min, VAD_var_param.Next_Min);
111if (mod(VAD_var_param.frm_count, single(8)) == 0)
112    VAD_var_param.Min_buffer = [VAD_var_param.Min_buffer(2:end), VAD_var_param.Next_Min];
113    VAD_var_param.Prev_Min = min(VAD_var_param.Min_buffer);
114    VAD_var_param.Next_Min = single(Inf);
115end
116
117if (VAD_var_param.frm_count < VAD_cst_param.Ni)
118    %% Initialization of running averages if frame number is less than 32
119    if (Ef < 21)
120        VAD_var_param.less_count = VAD_var_param.less_count + 1;
121        marker = NOISE;
122    else
123        % include only the frames that have an energy Ef greater than 21
124        marker = VOICE;
125        NE = (VAD_var_param.frm_count - 1) - VAD_var_param.less_count;
126    
127        VAD_var_param.MeanE = (VAD_var_param.MeanE * NE + Ef) / (NE+1);
128        VAD_var_param.MeanSZC = (VAD_var_param.MeanSZC * NE + ZC) / (NE+1);
129        VAD_var_param.MeanLSF = (VAD_var_param.MeanLSF * NE + LSF) / (NE+1);
130    end
131
132else 
133    %% Start calculating the chararcteristic energies of background noise
134    if (VAD_var_param.frm_count == VAD_cst_param.Ni)
135        VAD_var_param.MeanSE = VAD_var_param.MeanE - 10;
136        VAD_var_param.MeanSLE = VAD_var_param.MeanE - 12;
137    end
138    
139    % Difference measures between current frame parameters and running
140    % averages of background noise characteristics
141    dSE = VAD_var_param.MeanSE - Ef;
142    dSLE = VAD_var_param.MeanSLE - El;
143    dSZC = VAD_var_param.MeanSZC - ZC;
144
145    %% Initial VAD decision
146    if (Ef < 21)
147        marker = NOISE;
148    else
149        marker = vad_decision(dSLE, dSE, SD, dSZC);
150    end
151
152    v_flag = single(0);
153    
154    %% Voice activity decision smoothing
155    % from energy considerations and neighbouring past frame decisions
156    
157    % Step 1
158    if ((VAD_var_param.prev_markers(1) == VOICE) && (marker == NOISE) ...
159        && (Ef > VAD_var_param.MeanSE + 2) && (Ef > 21))
160        marker = VOICE;
161        v_flag = single(1);
162    end
163    
164    % Step 2
165    if (VAD_var_param.flag == 1)
166        if ((VAD_var_param.prev_markers(2) == VOICE) ...
167            && (VAD_var_param.prev_markers(1) == VOICE) ...
168            && (marker == NOISE) ...
169            && (abs(Ef - VAD_var_param.prev_energy) <= 3))
170        
171            VAD_var_param.count_ext = VAD_var_param.count_ext + 1;
172            marker = VOICE;
173            v_flag = single(1);
174           
175            if (VAD_var_param.count_ext <= 4)
176                VAD_var_param.flag = single(1);
177            else
178                VAD_var_param.count_ext = single(0);
179                VAD_var_param.flag = single(0);
180            end
181        end
182    else
183        VAD_var_param.flag = single(1);
184    end
185
186    if (marker == NOISE)
187        VAD_var_param.count_sil = VAD_var_param.count_sil + 1;
188    end
189  
190    % Step 3    
191    if ((marker == VOICE) && (VAD_var_param.count_sil > 10) ...
192        && (Ef - VAD_var_param.prev_energy <= 3))
193        marker = NOISE;
194        VAD_var_param.count_sil = single(0);
195    end
196    
197    if (marker == VOICE)
198        VAD_var_param.count_sil = single(0);
199    end
200
201    % Step 4
202    if ((Ef < VAD_var_param.MeanSE + 3) && (VAD_var_param.frm_count > VAD_cst_param.No) ...
203        && (v_flag == 0) && (rc(2) < 0.6))
204        marker = NOISE;
205    end
206    
207    %% Update running averages only in the presence of background noise
208    
209    if ((Ef < VAD_var_param.MeanSE + 3) && (rc(2) < 0.75) && (SD < 0.002532959))
210        VAD_var_param.count_update = VAD_var_param.count_update + 1;
211        % Modify update speed coefficients
212        if (VAD_var_param.count_update < VAD_cst_param.INIT_COUNT)
213            COEF = single(0.75);
214            COEFZC = single(0.8);
215            COEFSD = single(0.6);
216        elseif (VAD_var_param.count_update < VAD_cst_param.INIT_COUNT + 10)
217            COEF = single(0.95);
218            COEFZC = single(0.92);
219            COEFSD = single(0.65);
220        elseif (VAD_var_param.count_update < VAD_cst_param.INIT_COUNT + 20)
221            COEF = single(0.97);
222            COEFZC = single(0.94);
223            COEFSD = single(0.70);
224        elseif (VAD_var_param.count_update < VAD_cst_param.INIT_COUNT + 30)
225            COEF = single(0.99);
226            COEFZC = single(0.96);
227            COEFSD = single(0.75);
228        elseif (VAD_var_param.count_update < VAD_cst_param.INIT_COUNT + 40)
229            COEF = single(0.995);
230            COEFZC = single(0.99);
231            COEFSD = single(0.75);
232        else
233            COEF = single(0.995);
234            COEFZC = single(0.998);
235            COEFSD = single(0.75);
236        end
237
238        % Update mean of parameters LSF, SE, SLE, SZC
239        VAD_var_param.MeanSE = COEF * VAD_var_param.MeanSE + (1-COEF) * Ef;
240        VAD_var_param.MeanSLE = COEF * VAD_var_param.MeanSLE + (1-COEF) * El;
241        VAD_var_param.MeanSZC = COEFZC * VAD_var_param.MeanSZC + (1-COEFZC) * ZC;
242        VAD_var_param.MeanLSF = COEFSD * VAD_var_param.MeanLSF + (1-COEFSD) * LSF;        
243    end
244
245    if ((VAD_var_param.frm_count > VAD_cst_param.No) && ((VAD_var_param.MeanSE < Min) &&  (SD < 0.002532959)) || (VAD_var_param.MeanSE > Min + 10 ))
246        VAD_var_param.MeanSE = Min;
247        VAD_var_param.count_update = single(0);
248    end
249end
250
251%% Update parameters for next frame
252VAD_var_param.prev_energy = Ef;
253VAD_var_param.prev_markers = [marker, VAD_var_param.prev_markers(1)];
254
255idx = VAD_cst_param.L_FRAME + 1;
256VAD_var_param.window_buffer = speech_buf(idx:end);
257
258%% Return final decision
259vad_flag = marker;
260
261return
262
